#find the mmajority of the element 
arr = [3, 3, 4, 2, 4, 4, 2, 4, 4]

candidate = None
count = 0

for num in arr:
    if count == 0:
        candidate = num
    
    if num == candidate:
        count += 1
    else:
        count -= 1


if arr.count(candidate) > len(arr) // 2:
    print("Majority element is:", candidate)
else:
    print("No majority element")

arr = [1, 2, 3, 4]
n = len(arr)

result = [1] * n

left = 1
for i in range(n):
    result[i] = left
    left = left * arr[i]


right = 1
for i in range(n-1, -1, -1):
    result[i] = result[i] * right
    right = right * arr[i]

print(result)
